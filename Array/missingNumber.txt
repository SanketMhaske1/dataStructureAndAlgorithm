// Best Solution

void moveZeroes(vector<int>& nums) {

        int n = nums.size();

        for(int i=0;i<n;i++){
            for(int j=i+1;j<n;j++){
                if(nums[j-1]==0){
                    swap(nums[j],nums[j-1]);
                }
            }
        }
       
        
    }

Time Complexity = O(n)
Space Complexity = O(1)
